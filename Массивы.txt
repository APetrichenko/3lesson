 public static String[] expandArray(String[] inArray, int expandBy) {
        int initLength = inArray.length;
        String[] result = new String[initLength+expandBy];
        for (int i = 0; i <inArray.length;i++){
            result[i] = inArray[i];
        }
        return result;
    }
